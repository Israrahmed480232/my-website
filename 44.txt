<!-- Basic Calculator -->
<div class="box">
  <h2>Basic Calculator</h2>
  <input type="number" id="num1" placeholder="Enter first number">
  <input type="number" id="num2" placeholder="Enter second number"><br>
  <button onclick="calculate('+')">Add (+)</button>
  <button onclick="calculate('-')">Subtract (-)</button>
  <button onclick="calculate('*')">Multiply (ร)</button>
  <button onclick="calculate('/')">Divide (รท)</button><br>
  <p id="result"></p>
</div>

<!-- Area of Circle -->
<div class="box">
  <h2>Area of Circle</h2>
  <input type="number" id="radius" placeholder="Enter radius">
  <button onclick="areaCircle()">Calculate</button>
  <p id="circleResult"></p>
</div>

<!-- Pythagoras Theorem -->
<div class="box">
  <h2>Pythagoras Theorem (Find Hypotenuse)</h2>
  <input type="number" id="sideA" placeholder="Enter side A">
  <input type="number" id="sideB" placeholder="Enter side B">
  <button onclick="pythagoras()">Calculate</button>
  <p id="pythagorasResult"></p>
</div>

<script>
  function calculate(operator) {
    let n1 = parseFloat(document.getElementById('num1').value);
    let n2 = parseFloat(document.getElementById('num2').value);
    let res = 0;
    if (operator === '+') res = n1 + n2;
    else if (operator === '-') res = n1 - n2;
    else if (operator === '*') res = n1 * n2;
    else if (operator === '/') {
      if (n2 === 0) {
        document.getElementById('result').innerText = "Error: Division by zero";
        return;
      }
      res = n1 / n2;
    }
    document.getElementById('result').innerText = "Result: " + res;
  }

  function areaCircle() {
    let r = parseFloat(document.getElementById('radius').value);
    let area = 3.1416 * r * r;
    document.getElementById('circleResult').innerText = "Area: " + area.toFixed(2);
  }

  function pythagoras() {
    let a = parseFloat(document.getElementById('sideA').value);
    let b = parseFloat(document.getElementById('sideB').value);
    let c = Math.sqrt(a * a + b * b);
    document.getElementById('pythagorasResult').innerText = "Hypotenuse: " + c.toFixed(2);
  }
</script>
