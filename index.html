import java.util.Scanner; public class OnlineEducation { public static void main(String[] args) { Scanner input = new Scanner(System.in); String[] courses = {"Math", "Physics", "Chemistry", "Biology", "Computer Science"}; String studentName; System.out.println("===== Welcome to Online Education Platform ====="); System.out.print("Enter your name: "); studentName = input.nextLine(); System.out.println("\nHello, " + studentName + "! Available courses:"); for (int i = 0; i < courses.length; i++) { System.out.println((i + 1) + ". " + courses[i]); } System.out.print("\nSelect a course by entering the number: "); int choice = input.nextInt(); if (choice >= 1 && choice <= courses.length) { System.out.println("\nYou have successfully registered for " + courses[choice - 1] + " course."); System.out.println("Best of luck, " + studentName + "!"); } else { System.out.println("\nInvalid choice. Please restart and select a valid course."); } System.out.println("\n===== Thank you for using Online Education Platform ====="); input.close(); } }
